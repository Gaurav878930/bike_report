#  SQL Business Problems & Solutions

## 1. Top Selling Products

**Objective:** Find the top 10 products based on total quantity sold.

```sql
SELECT
    p.product_id,
    p.product_name,
    SUM(oi.quantity) AS total_quantity
FROM product p
JOIN order_item oi
ON p.product_id = oi.product_id
GROUP BY p.product_id, p.product_name
ORDER BY total_quantity DESC
LIMIT 10;
```

---

## 2. Revenue by Category

**Objective:** Calculate total revenue generated by each category after discounts.

```sql
SELECT
    c.category_name,
    ROUND(
        SUM(
            oi.quantity * oi.list_price * (1 - oi.discount)
        ),
        2
    ) AS revenue
FROM product p
JOIN categories c
ON p.category_id = c.category_id
JOIN order_item oi
ON p.product_id = oi.product_id
GROUP BY c.category_name
ORDER BY revenue DESC;
```

---

## 3. Store-wise Sales Performance

**Objective:** Rank stores by total revenue.

```sql
SELECT
    s.store_name,
    ROUND(
        SUM(
            oi.quantity * oi.list_price * (1 - oi.discount)
        ),
        2
    ) AS revenue,
    RANK() OVER (
        ORDER BY
        SUM(
            oi.quantity * oi.list_price * (1 - oi.discount)
        ) DESC
    ) AS store_rank
FROM store s
JOIN orders o
ON s.store_id = o.store_id
JOIN order_item oi
ON o.order_id = oi.order_id
GROUP BY s.store_name;
```

## 4. Customer Order Analysis

**Objective:** Find customers with more than 5 orders and calculate their spending.

```sql
SELECT
    c.first_name,
    c.last_name,
    COUNT(DISTINCT o.order_id) AS total_orders,
    ROUND(
        SUM(
            oi.quantity * oi.list_price * (1 - oi.discount)
        ),
        2
    ) AS total_spending
FROM customer c
JOIN orders o
ON c.customer_id = o.customer_id
JOIN order_item oi
ON o.order_id = oi.order_id
GROUP BY c.first_name, c.last_name
HAVING COUNT(DISTINCT o.order_id) > 5;
```

## 5. Staff Performance

**Objective:** Identify staff members generating the highest revenue.

```sql
SELECT
    s.first_name,
    s.last_name,
    COUNT(DISTINCT o.order_id) AS total_orders,
    ROUND(
        SUM(
            oi.quantity * oi.list_price * (1 - oi.discount)
        ),
        2
    ) AS revenue
FROM staff s
JOIN orders o
ON s.staff_id = o.staff_id
JOIN order_item oi
ON o.order_id = oi.order_id
GROUP BY s.first_name, s.last_name
ORDER BY revenue DESC;
```

## 6. Best Selling Product in Each Category

```sql
WITH product_sales AS (
    SELECT
        c.category_name,
        p.product_name,
        SUM(oi.quantity) AS total_quantity,
        DENSE_RANK() OVER (
            PARTITION BY c.category_name
            ORDER BY SUM(oi.quantity) DESC
        ) AS rank
    FROM order_item oi
    JOIN product p
    ON oi.product_id = p.product_id
    JOIN categories c
    ON p.category_id = c.category_id
    GROUP BY c.category_name, p.product_name
)

SELECT *
FROM product_sales
WHERE rank = 1;
```

## 7. Monthly Revenue Growth by Store

```sql
WITH monthly_sales AS (
    SELECT
        s.store_name,
        DATE_TRUNC('month', o.order_date) AS month,
        SUM(
            oi.quantity * oi.list_price * (1 - oi.discount)
        ) AS revenue
    FROM store s
    JOIN orders o
    ON s.store_id = o.store_id
    JOIN order_item oi
    ON o.order_id = oi.order_id
    GROUP BY s.store_name, month
)

SELECT
    *,
    LAG(revenue)
    OVER(PARTITION BY store_name ORDER BY month)
    AS previous_month
FROM monthly_sales;
```

## 8. Inventory Risk Analysis

```sql
SELECT
    p.product_name,
    c.category_name,
    SUM(s.quantity) AS current_stock
FROM product p
JOIN stock s
ON p.product_id = s.product_id
JOIN categories c
ON p.category_id = c.category_id
GROUP BY p.product_name, c.category_name;
```

## 9. Customer Lifetime Value (CLV)

```sql
SELECT
    c.customer_id,
    c.first_name,
    ROUND(
        SUM(
            oi.quantity * oi.list_price * (1 - oi.discount)
        ),
        2
    ) AS customer_lifetime_value
FROM customer c
JOIN orders o
ON c.customer_id = o.customer_id
JOIN order_item oi
ON o.order_id = oi.order_id
GROUP BY c.customer_id, c.first_name
ORDER BY customer_lifetime_value DESC
LIMIT 10;
```

## 10. Store & Category Revenue Contribution

```sql
WITH category_revenue AS (
SELECT
    st.store_name,
    c.category_name,
    SUM(
        oi.quantity * oi.list_price * (1 - oi.discount)
    ) AS revenue
FROM store st
JOIN orders o
ON st.store_id = o.store_id
JOIN order_item oi
ON o.order_id = oi.order_id
JOIN product p
ON oi.product_id = p.product_id
JOIN categories c
ON p.category_id = c.category_id
GROUP BY st.store_name, c.category_name
)

SELECT
    *,
    ROUND(
        revenue * 100 /
        SUM(revenue) OVER(PARTITION BY store_name),
        2
    ) AS revenue_percentage
FROM category_revenue;
```
